import { Playground } from 'dokz'
import { useInjectedConnector, withWeb3React } from '@rainbowkit/core'
import { Box, Button, Heading, Code } from '@chakra-ui/react'
import { formatEther } from '@ethersproject/units'

# Index

## Hooks

### Connector hooks

#### `useInjectedConnector`

<Playground>
  {withWeb3React(() => {
    const { isConnected, connect, disconnect, address, provider } = useInjectedConnector()
    const [bal, set] = React.useState('')
    React.useEffect(() => {
      if (provider && address) provider.getBalance(address).then((b) => set(formatEther(b).slice(0, 10)))
    }, [provider, address])
    return (
      <Box p="2rem">
        {isConnected ? (
          <div>
            <Button onClick={disconnect}>Disconnect</Button>
          </div>
        ) : (
          <div>
            <Button onClick={connect}>Connect</Button>
          </div>
        )}
        <Heading>Address</Heading>
        <Box>
          Address: <Code>{address}</Code>
        </Box>
        <Box>
          Balance: <Code>{bal}</Code>
        </Box>
      </Box>
    )
  })}
</Playground>
